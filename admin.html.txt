    <h2>生成一次性测试链接</h2>
    <label>管理员密码</label>
    <input type="password" id="password" placeholder="输入密码">
    <label>生成数量</label>
    <input type="number" id="count" value="10" min="1" max="200">
    <button onclick="generate()">生成链接</button>
    <div id="output"></div>

    <script>
        async function generate() {
            const password = document.getElementById('password').value;
            const count = document.getElementById('count').value;
            const output = document.getElementById('output');
            output.innerHTML = '生成中...';
            try {
                const res = await fetch('/api/generate', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ password, count: parseInt(count) })
                });
                const data = await res.json();
                if (res.ok) {
                    output.innerHTML = '<h3>生成的链接（每个只能用一次）：</h3><pre>' + data.links.join('\n') + '</pre>';
                } else {
                    output.innerHTML = `<p class="error">错误：${data.error}</p>`;
                }
            } catch (err) {
                output.innerHTML = '<p class="error">请求失败，请检查网络或后端配置。</p>';
            }
        }
    </script>
